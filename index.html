require('dotenv').config(); const express = require('express'); const axios = require('axios'); const admin = require('firebase-admin'); const nodemailer = require('nodemailer'); const cors = require('cors'); const Fingerprint = require('express-fingerprint'); const jwt = require('jsonwebtoken'); const app = express(); app.use(cors({ origin: process.env.FRONTEND_URL, credentials: true })); app.use(express.json()); app.use(Fingerprint()); let db; try { let serviceAccount; if (process.env.SERVICE_ACCOUNT_KEY_JSON) { serviceAccount = JSON.parse(process.env.SERVICE_ACCOUNT_KEY_JSON); } else { serviceAccount = require('./serviceAccountKey.json'); } admin.initializeApp({ credential: admin.credential.cert(serviceAccount) }); db = admin.firestore(); console.log("Firebase Admin SDK successfully initialized!"); } catch (e) { console.error("FATAL ERROR initializing Firebase:", e.message); process.exit(1); } // Helper function to get client IP const getIp = (req) => req.headers['x-forwarded-for'] || req.connection.remoteAddress; // Middleware to verify JWT const verifyToken = (req, res, next) => { const token = req.headers['x-access-token']; if (!token) return res.status(403).json({ message: "No token provided." }); jwt.verify(token, process.env.JWT_SECRET, (err, decoded) => { if (err) return res.status(401).json({ message: "Unauthorized." }); req.userId = decoded.id; next(); }); }; app.get('/', (req, res) => { res.status(200).send('Server is healthy and running!'); }); app.get('/auth/discord', (req, res) => { const discordAuthUrl = `https://discord.com/api/oauth2/authorize?client_id=${process.env.DISCORD_CLIENT_ID}&redirect_uri=${encodeURIComponent(process.env.SERVER_URL + '/auth/discord/callback')}&response_type=code&scope=identify%20email`; res.redirect(discordAuthUrl); }); app.get('/auth/discord/callback', async (req, res) => { const { code } = req.query; if (!code) return res.status(400).send('Error: Authorization code is missing.'); try { const tokenResponse = await axios.post('https://discord.com/api/oauth2/token', new URLSearchParams({ client_id: process.env.DISCORD_CLIENT_ID, client_secret: process.env.DISCORD_CLIENT_SECRET, grant_type: 'authorization_code', code, redirect_uri: process.env.SERVER_URL + '/auth/discord/callback' })); const { access_token } = tokenResponse.data; const userResponse = await axios.get('https://discord.com/api/users/@me', { headers: { 'Authorization': `Bearer ${access_token}` } }); const discordUser = userResponse.data; const userRef = db.collection('users').doc(discordUser.id); const ip = getIp(req); await userRef.set({ username: discordUser.username, avatar: discordUser.avatar, email: discordUser.email, login_history: admin.firestore.FieldValue.arrayUnion({ ip, timestamp: new Date() }) }, { merge: true }); const token = jwt.sign({ id: discordUser.id }, process.env.JWT_SECRET, { expiresIn: 86400 }); // 24 hours res.redirect(`${process.env.FRONTEND_URL}?token=${token}`); } catch (error) { console.error("Error during Discord authentication:", error.response ? error.response.data : error.message); res.status(500).send('An error occurred during authentication.'); } }); app.get('/api/verify-token', verifyToken, async (req, res) => { try { const userDoc = await db.collection('users').doc(req.userId).get(); if (!userDoc.exists) return res.status(404).json({ message: "User not found." }); const { username, avatar } = userDoc.data(); res.status(200).json({ userId: req.userId, username, avatar }); } catch(e) { res.status(500).json({ message: "Server error." }); } }); app.post('/api/send-email', async (req, res) => { const { userId, recipients, subject, message, sendAs, fromName } = req.body; const ip = getIp(req); const fingerprint = req.fingerprint.hash; if(!recipients || !subject || !message || !sendAs) return res.status(400).json({ message: "Missing required data." }); const now = new Date(); const twentyFourHoursAgo = new Date(now.getTime() - (24 * 60 * 60 * 1000)); // Guest Logic if (!userId || userId === 'guest') { if (sendAs === 'site') { const usageRef = db.collection('guest_usage'); const query = usageRef.where('fingerprint', '==', fingerprint).where('timestamp', '>=', twentyFourHoursAgo); const snapshot = await query.get(); if (snapshot.size >= 10) { return res.status(429).json({ message: "Guest daily limit (10) for Site Email reached. Please try again tomorrow or log in." }); } } } // Registered User Logic if (userId && userId !== 'guest') { if (sendAs === 'site') { const usageRef = db.collection('user_usage'); const query = usageRef.where('userId', '==', userId).where('timestamp', '>=', twentyFourHoursAgo); const snapshot = await query.get(); if (snapshot.size >= 50) { return res.status(429).json({ message: "You have reached your daily limit of 50 site emails. Please try again tomorrow." }); } } } try { let transporter, fromAddress, finalFromName; if (userId && userId !== 'guest') { const userDoc = await db.collection('users').doc(userId).get(); finalFromName = fromName || userDoc.data().username; } else { finalFromName = fromName || 'Guest'; } if (sendAs === 'user') { // Logic for guest sending with their own gmail const { gmailAddress, appPassword } = req.body; if (!gmailAddress || !appPassword) return res.status(400).json({ message: "Gmail credentials are required for guests." }); transporter = nodemailer.createTransport({ service: 'gmail', auth: { user: gmailAddress, pass: appPassword } }); fromAddress = gmailAddress; } else { // sendAs === 'site' transporter = nodemailer.createTransport({ host: process.env.SMTP_HOST, port: parseInt(process.env.SMTP_PORT, 10), auth: { user: process.env.SMTP_USER, pass: process.env.SMTP_PASS } }); fromAddress = process.env.SITE_SENDER_EMAIL; } await transporter.sendMail({ from: `"${finalFromName}" <${fromAddress}>`, to: recipients, subject: subject, text: message }); // Log usage for guests and users for site email if (sendAs === 'site') { const logCollection = userId && userId !== 'guest' ? 'user_usage' : 'guest_usage'; await db.collection(logCollection).add({ userId: userId || 'guest', fingerprint, ip, timestamp: now }); } res.status(200).json({ message: "Emails have been sent successfully!" }); } catch (error) { console.error("Error sending email:", error); res.status(500).json({ message: "A server error occurred while sending emails." }); } }); const PORT = process.env.PORT || 3000; app.listen(PORT, '0.0.0.0', () => { console.log(`Server is running on port ${PORT}`); });
